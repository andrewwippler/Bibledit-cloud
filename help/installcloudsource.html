translate("Follow the steps below to install Bibledit Cloud.")


Step 1. translate("Open a terminal.")


Step 2. translate("Update the software sources.")

Debian, Ubuntu sudo apt update
CentOS sudo dnf update

Step 3. translate("Install and enable extra package repositories.")

CentOS sudo dnf install epel-release

Step 4. translate("Install the software Bibledit relies on:")

Debian, Ubuntu sudo apt install build-essential autotools-dev autoconf autoconf-archive git zip pkgconf libcurl4-openssl-dev libssl-dev poppler-utils libsword-utils diatheke ldap-utils libmimetic-dev certbot libxml2-dev libutf8proc-dev libgumbo-dev libtidy-dev libpugixml-dev
CentOS sudo dnf install autoconf automake gcc gcc-c++ libtool git zip unzip pkgconfig libcurl-devel openssl-devel curl make poppler-utils libxml2-devel utf8proc gumbo-parser-devel libtidy-devel pugixml-devel

Step 5. translate("Go to the home directory.")

Debian, Ubuntu, CentOS cd

Step 6. translate("Download a recent version of Bibledit from") GitHub.


Step 7. translate("Unpack the downloaded Bibledit in folder bibledit.")

Debian, Ubuntu, CentOS

mkdir -p bibledit

tar xf bibledit-5.1.xxx.tar.gz -C bibledit --strip-components=1

translate("Change the xxx above to the version number of Bibledit that was downloaded.")


Step 8. translate("Change to the directory.")

Debian, Ubuntu, CentOS

cd bibledit


Step 9. translate("Build Bibledit.")

Debian, Ubuntu, CentOS

./configure

make

translate("This takes a while.") translate("There will be many technical messages, but there should be no errors.")


Step 10. translate("Open port 8080 in the firewall.")

CentOS

sudo firewall-cmd --zone=public --add-port=8080/tcp --permanent

sudo firewall-cmd --reload


Step 11. translate("Start Bibledit.")

Debian, Ubuntu, CentOS

./bibledit &
